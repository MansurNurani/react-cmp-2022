## Install

```bash
npm install --save react-cmp-2022
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-cmp-2022'
import 'react-cmp-2022/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
